<!DOCTYPE html>

<html>

<head>

    <title>Reaction Time Game</title>

    <script>

        async function playGame() {

            const velocity = parseFloat(document.getElementById("velocity").value) || 0;

            const acceleration = parseFloat(document.getElementById("acceleration").value) || 0;

            const winStreak = parseInt(document.getElementById("win_streak").value) || 0;

            const lossStreak = parseInt(document.getElementById("loss_streak").value) || 0;

            const opponentLastSound = parseFloat(document.getElementById("opponent_last_sound").value) || 0;

            const opponentVelocity = parseFloat(document.getElementById("opponent_velocity").value) || 0;

            const opponentAcceleration = parseFloat(document.getElementById("opponent_acceleration").value) || 0;


            const response = await fetch('/bot_decision', {

                method: 'POST',

                headers: {

                    'Content-Type': 'application/json',

                },

                body: JSON.stringify({

                    velocity,

                    acceleration,

                    win_streak: winStreak,

                    loss_streak: lossStreak,

                    opponent_last_sound: opponentLastSound,

                    opponent_velocity: opponentVelocity,

                    opponent_acceleration: opponentAcceleration,

                }),

            });


            const data = await response.json();

            document.getElementById("result").innerText = `Bot's chosen blast level: ${data.bot_blast}`;

        }

    </script>

</head>

<body>

    <h1>Reaction Time Game</h1>

    <p>Enter the game dynamics below:</p>

    <label>Velocity:</label> <input id="velocity" type="number"><br>

    <label>Acceleration:</label> <input id="acceleration" type="number"><br>

    <label>Win Streak:</label> <input id="win_streak" type="number"><br>

    <label>Loss Streak:</label> <input id="loss_streak" type="number"><br>

    <label>Opponent Last Sound:</label> <input id="opponent_last_sound" type="number"><br>

    <label>Opponent Velocity:</label> <input id="opponent_velocity" type="number"><br>

    <label>Opponent Acceleration:</label> <input id="opponent_acceleration" type="number"><br>

    <button onclick="playGame()">Submit</button>

    <p id="result"></p>

</body>
